# Run this file with $ python -m doctest -v doctest_linkify.txt

>>> from mklists.util import linkify

>>> linkify('A line with <a href="http://example.org">http://example.org</a>.\n')
'A line with <a href="http://example.org">http://example.org</a>.\n'

>>> linkify('http://example.org')
'<a href="http://example.org">http://example.org</a>'

# Shows that function does not test for non-sensical URLs.
>>> linkify('http://...')
'<a href="http://...">http://...</a>'

>>> linkify('http://192.168.2.1/html/login/status.html?lang=en')
'<a href="http://192.168.2.1/html/login/status.html?lang=en">http://192.168.2.1/html/login/status.html?lang=en</a>'

>>> linkify('http://192.168.56.100:8888/')
'<a href="http://192.168.56.100:8888/">http://192.168.56.100:8888/</a>'

>>> linkify('http://carnegieeurope.eu/strategiceurope/?fa=3D64063=')
'<a href="http://carnegieeurope.eu/strategiceurope/?fa=3D64063=">http://carnegieeurope.eu/strategiceurope/?fa=3D64063=</a>'

>>> linkify('http://ctrlpvim.github.io/ctrlp.vim/#installation')
'<a href="http://ctrlpvim.github.io/ctrlp.vim/#installation">http://ctrlpvim.github.io/ctrlp.vim/#installation</a>'

>>> linkify('http://en.wikipedia.org/wiki/Talk:Dublin_Core')
'<a href="http://en.wikipedia.org/wiki/Talk:Dublin_Core">http://en.wikipedia.org/wiki/Talk:Dublin_Core</a>'

>>> linkify('http://en.wikipedia.org/wiki/USS_Rizzi_%28DE-537%29')
'<a href="http://en.wikipedia.org/wiki/USS_Rizzi_%28DE-537%29">http://en.wikipedia.org/wiki/USS_Rizzi_%28DE-537%29</a>'

>>> linkify('http://explore.dublincore.net/rdf/lrmi/#/resource')
'<a href="http://explore.dublincore.net/rdf/lrmi/#/resource">http://explore.dublincore.net/rdf/lrmi/#/resource</a>'

>>> linkify('http://twitter.com/#!/search/%23dcmi11')
'<a href="http://twitter.com/#!/search/%23dcmi11">http://twitter.com/#!/search/%23dcmi11</a>'

# This application will not accept commas as valid URL characters.
>>> linkify('http://standorte.deutschepost.de,')
'<a href="http://standorte.deutschepost.de">http://standorte.deutschepost.de</a>,'

>>> linkify('https://www.w3.org/2013/data/CG/wiki')
'<a href="https://www.w3.org/2013/data/CG/wiki">https://www.w3.org/2013/data/CG/wiki</a>'

>>> linkify('see info (https://www.w3.org/2013/data/CG/wiki)')
'see info (<a href="https://www.w3.org/2013/data/CG/wiki">https://www.w3.org/2013/data/CG/wiki</a>)'

>>> linkify('see info (https://example1.org), http://example2.org')
'see info (<a href="https://example1.org">https://example1.org</a>), <a href="http://example2.org">http://example2.org</a>'
